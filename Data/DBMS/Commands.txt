Creating given table customer:- 
-------------------------------
CREATE TABLE customer(
CUST_ID NUMBER,
 CUST_FIRST_NAME VARCHAR2(20),
 CUST_LAST_NAME VARCHAR2(20),
 TERRITORY VARCHAR2(15),
 CREDIT_LIMIT NUMBER(6,2),
 MGR_ID NUMBER,
 DOB DATE,
 MARITORIAL_STATUS VARCHAR2(20),
 G VARCHAR2(1),
 INCOME NUMBER
);

-------------------------------
